Java by Example: HTTP Client

Java 11+ includes a modern HttpClient in the standard library.

 
import java.net.URI;
import java.net.http.*;
import java.net.http.HttpResponse.BodyHandlers;

public class HttpClientExample {
    public static void main(String[] args) throws Exception {
        var client = HttpClient.newHttpClient();

Simple GET request.

 
        // GET request
        var request = HttpRequest.newBuilder()
            .uri(URI.create("https://httpbin.org/get"))
            .build();

        var response = client.send(request,
            BodyHandlers.ofString());
        System.out.println("status: " + response.statusCode());
        System.out.println("body length: " + response.body().length());

POST request with a body.

 
        // POST request
        var postReq = HttpRequest.newBuilder()
            .uri(URI.create("https://httpbin.org/post"))
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers
                .ofString("{\"key\": \"value\"}"))
            .build();

        var postRes = client.send(postReq,
            BodyHandlers.ofString());
        System.out.println("post status: " + postRes.statusCode());
    }
}
