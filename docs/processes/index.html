Java by Example: Processes

Java can spawn external processes using ProcessBuilder.

 
import java.io.*;

public class Processes {
    public static void main(String[] args) throws Exception {
        // Simple command
        var process = new ProcessBuilder("date")
            .start();
        var output = new String(
            process.getInputStream().readAllBytes());
        System.out.println("output: " + output.trim());
        System.out.println("exit: " + process.waitFor());

Running commands with arguments and handling stderr.

 
        // Command with arguments
        var p2 = new ProcessBuilder("ls", "-la", "/tmp")
            .redirectErrorStream(true)
            .start();
        var lines = new BufferedReader(
            new InputStreamReader(p2.getInputStream()))
            .lines().limit(3).toList();
        lines.forEach(System.out::println);
        p2.waitFor();
    }
}
