Java by Example: Writing Files

Writing files in Java follows similar patterns to reading them.

 
import java.io.*;
import java.nio.file.*;

public class WritingFiles {
    public static void main(String[] args) throws Exception {
        // Write a string directly (Java 11+)
        Files.writeString(Path.of("/tmp/dat1"), "hello\njava\n");

Writing with more control using BufferedWriter.

 
        // Buffered writing
        try (var writer = Files.newBufferedWriter(
                Path.of("/tmp/dat2"))) {
            writer.write("buffered\n");
            writer.write("writes\n");
        } // auto-flushes and closes

Appending to files.

 
        // Append to an existing file
        Files.writeString(
            Path.of("/tmp/dat1"),
            "appended\n",
            StandardOpenOption.APPEND);

        // Verify
        System.out.print(Files.readString(Path.of("/tmp/dat1")));
        System.out.print(Files.readString(Path.of("/tmp/dat2")));
    }
}
 
$ java WritingFiles.java
hello
java
appended
buffered
writes
