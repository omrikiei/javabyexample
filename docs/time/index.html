Java by Example: Time

Java 8+ provides the java.time package for comprehensive date/time support.

 
import java.time.*;
import java.time.format.*;
import java.time.temporal.*;

public class Time {
    public static void main(String[] args) {
        var now = Instant.now();
        System.out.println("now: " + now);

Creating specific times and dates.

 
        var dt = LocalDateTime.of(2023, 6, 15, 14, 30, 0);
        System.out.println("date: " + dt);
        System.out.println("year: " + dt.getYear());
        System.out.println("month: " + dt.getMonth());
        System.out.println("day: " + dt.getDayOfMonth());

Duration and period calculations.

 
        var earlier = LocalDateTime.of(2023, 1, 1, 0, 0);
        var diff = Duration.between(earlier, dt);
        System.out.println("diff: " + diff);
        System.out.println("hours: " + diff.toHours());

Formatting and parsing.

 
        var fmt = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm");
        System.out.println("formatted: " + dt.format(fmt));

        var parsed = LocalDateTime.parse(
            "2023-06-15 14:30", fmt);
        System.out.println("parsed: " + parsed);
    }
}
