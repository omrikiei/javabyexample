Java by Example: Regular Expressions

Java provides built-in support for regular expressions via the java.util.regex package.

 
import java.util.regex.*;

public class RegularExpressions {
    public static void main(String[] args) {
        // Test whether a pattern matches a string
        boolean match = Pattern.matches("p([a-z]+)ch", "peach");
        System.out.println(match);

Compile a pattern for reuse.

 
        var p = Pattern.compile("p([a-z]+)ch");

        // Find a match
        System.out.println(p.matcher("peach punch").find());

        // Find the matching string
        var m = p.matcher("peach punch");
        if (m.find()) {
            System.out.println("match: " + m.group());
        }

Find all matches.

 
        // Find all matches
        m = p.matcher("peach punch pinch");
        while (m.find()) {
            System.out.print(m.group() + " ");
        }
        System.out.println();
    }
}
 
$ java RegularExpressions.java
